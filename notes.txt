import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from netCDF4 import Dataset
import matplotlib as mpl

# Load the netCDF file
file_path = 'path_to_your_file.nc'
cdf = Dataset(file_path)
# Extract variables
jPar = cdf.variables['jPar'][:][n,:]  # Replace n with the desired time index
cLat_deg = cdf.variables['cLat_deg'][:][n,:]
mlt_hr = cdf.variables['mlt_hr'][:][n,:]

# Create a grid for plotting
df = pd.DataFrame({'cLat_deg': cLat_deg.flatten(), 'mlt_hr': mlt_hr.flatten(), 'jPar': jPar.flatten()})
pivot_table = df.pivot_table(index='cLat_deg', columns='mlt_hr', values='jPar')

# Plotting
fig=plt.figure(figsize=(10, 6))
ax=fig.add_subplot(111, projection='polar')
ax.set_theta_zero_location('S')

r,th = np.meshgrid(pivot_table.index, np.linspace(0,2*np.pi,24, endpoint=False))
scale = mpl.colors.Normalize(vmin=-1.5, vmax=1.5) # Adjust vmin and vmax as needed this is what is default on AMPERE website plots
cbar = fig.colorbar(ax.pcolormesh(th, r, pivot_table.T, norm=scale, cmap='bwr', shading='auto'), ax=ax)
cbar.set_label('Field-aligned Current Density (mA/m²)')
ax.set_yticklabels([])  # Hide radial tick labels
ax.set_xticks(np.linspace(0, 2 * np.pi, 24, endpoint=False))
ax.set_xticklabels([f'{int(hr)}' for hr in np.linspace(0, 24, 24, endpoint=False)])
ax.set_title('Field-aligned Current Density Map', va='bottom')
plt.show()
